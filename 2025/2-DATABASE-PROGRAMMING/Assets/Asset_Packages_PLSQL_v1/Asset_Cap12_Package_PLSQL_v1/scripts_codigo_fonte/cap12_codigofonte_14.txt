SET SERVEROUTPUT ON;
DECLARE
	V_DISTANCIA_KM NUMBER := 9;
	V_DISTANCIA_MILHAS NUMBER;
BEGIN
	-- Aqui calculamos a distância de 9KM para milhas
	V_DISTANCIA_MILHAS := ROUND( V_DISTANCIA_KM * PKG_DB_COMERCIAL.V_CONVERTE_KM_MILHAS,1);
	
	DBMS_OUTPUT.PUT_LINE(V_DISTANCIA_KM || ' KM CORRESPONDE A ' || V_DISTANCIA_MILHAS || ' MILHAS');
END;
