SET SERVEROUTPUT ON;
DECLARE
V_QT_GRAMAS NUMBER(4) := 200;
V_QT_ONCAS  NUMBER;
BEGIN
    -- CHAMADA DA VARIÁVEL GLOBAL V_CONVERTE_GRAMA_ONCA
    -- QUE IRÁ TRAZER O VALOR FIXO A SER CONVERTIDO.
    V_QT_ONCAS := V_QT_GRAMAS * PKG_DB_COMERCIAL.V_CONVERTE_GRAMA_ONCA; 
    
    DBMS_OUTPUT.PUT_LINE(V_QT_GRAMAS || ' gramas equivalem a ' || V_QT_ONCAS || ' onças.');
END;
