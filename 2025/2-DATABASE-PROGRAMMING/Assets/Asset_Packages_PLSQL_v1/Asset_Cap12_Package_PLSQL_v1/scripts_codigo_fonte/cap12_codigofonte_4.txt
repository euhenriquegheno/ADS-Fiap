   v_converte_grama_onca CONSTANT NUMBER := 1 / 28.3495; -- 1 grama equivale a 0.03527396 onças

   -- Função para converter gramas para onças
   FUNCTION FUN_DB_GRAMAS_PARA_ONCAS(p_qt_gramas NUMBER) RETURN NUMBER;

   -- Procedure para exibir o resultado da conversão
   PROCEDURE PRC_DB_EXIBIR_CONVERSAO_GRAMA_ONCA(p_gramas NUMBER);
