-- CONSULTA QUE UNE PELO COMANDO UNION ALL
-- OS DADOS DAS 2 TABELAS
--
SELECT 'VALOR TOTAL PEDIDO:' MSG1, VL_TOT_PEDIDO
FROM DB_PEDIDO 
WHERE NR_LOJA =1 
AND NR_PEDIDO = 10
UNION ALL
SELECT  'VALOR TOTAL ITEM PEDIDO:' MSG,SUM(IP.QT_PEDIDO * IP.VL_UNITARIO) VAL_TOT_ITEM_PEDIDO
FROM    DB_ITEM_PEDIDO IP
WHERE IP.NR_LOJA =1 
AND IP.NR_PEDIDO = 10;
