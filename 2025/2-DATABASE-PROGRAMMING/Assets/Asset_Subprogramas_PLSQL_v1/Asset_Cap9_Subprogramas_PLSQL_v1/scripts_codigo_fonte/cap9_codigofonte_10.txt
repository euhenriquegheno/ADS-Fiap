CREATE OR REPLACE PROCEDURE PRC_DB_ATUALIZA_ESTRELAS_CLIENTE(
    P_NR_CLIENTE DB_CLIENTE.NR_CLIENTE%TYPE,
	P_QT_ESTRELAS DB_CLIENTE.QT_ESTRELAS%TYPE
) IS
BEGIN
    -- ATUALIZA A QUANTIDADE DE ESTRELAS DE UM DETERMINADO CLIENTE
    UPDATE DB_CLIENTE
    SET QT_ESTRELAS = P_QT_ESTRELAS
    WHERE NR_CLIENTE = P_NR_CLIENTE;
    -- Exibe o dado atualizado
    DBMS_OUTPUT.PUT_LINE('O CLIENTE DE NÚMERO (' || P_NR_CLIENTE || ') POSSUI AGORA ' || P_QT_ESTRELAS || ' ESTRELAS');
EXCEPTION
	-- CASO ENCONTRE ALGUM ERRO, ESTE SERÁ CAPTURADO E O PROCESSAMENTO SERÁ ENCERRADO.
    WHEN OTHERS THEN
        RAISE_APPLICATION_ERROR(-20005, 'ERRO CRÍTICO NA ATUALIZAÇÃO ' || SQLERRM);
END;
